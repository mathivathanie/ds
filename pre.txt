import pandas as pd
from sklearn.utils import resample

df=pd.read_csv("gold.csv")

print("Before balancing:")
print(df['Target'].value_counts())

majority=df[df['Target']==df['Target'].mode()[0]]
minority=df[df['Target']!=df['Target'].mode()[0]]

min_up=resample(
    minority,
    replace=True,
    n_samples=len(majority),
    random_state=42
)

df_bal=pd.concat([majority,min_up])

print("After:")
print(df_bal['Target'].value_counts())

df_bal.to_csv("gold_bal.csv",index=False)
